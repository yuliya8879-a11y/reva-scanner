```python
from aiogram import Router
from aiogram.filters import CommandStart
from aiogram.types import Message
from sqlalchemy.ext.asyncio import AsyncSession
from app.services.user_service import UserService

router = Router(name="start")

WELCOME_TEXT = (
    "Привет! Я <b>Глаз Бога</b> — AI-сканер вашего бизнеса.\n\n"
    "За 15 минут я найду:\n"
    "• Где утекают деньги\n"
    "• Слепые зоны в управлении\n"
    "• Где вы сами себе мешаете расти\n"
    "• Команда и риски\n"
    "• Нумерологический портрет бизнеса\n\n"
    "Бесплатный мини-скан — /scan\n"
    "Полный разбор за 3500₽ — после мини-скана."
)

@router.message(CommandStart())
async def cmd_start(message: Message, session_factory) -> None:
    async with session_factory() as session:
        await UserService.get_or_create(
            session=session,
            telegram_id=message.from_user.id,
            username=message.from_user.username,
            full_name=message.from_user.full_name,
        )
    await message.answer(WELCOME_TEXT)
```

**app/main.py** — FastAPI + aiogram 3 integration exactly as in RESEARCH.md Pattern 1:
```python
from contextlib import asynccontextmanager
from fastapi import FastAPI, Request
from aiogram import Bot, Dispatcher
from aiogram.client.default import DefaultBotProperties
from aiogram.enums import ParseMode
from aiogram.types import Update
from app.config import settings
from app.database import async_session_factory
from app.bot.router import router as bot_router

bot = Bot(
    token=settings.telegram_bot_token,
    default=DefaultBotProperties(parse_mode=ParseMode.HTML)
)
dp = Dispatcher()
dp.include_router(bot_router)

WEBHOOK_PATH = "/webhook/telegram"

@asynccontextmanager
async def lifespan(app: FastAPI):
    # Pass session factory to all handlers via dispatcher workflow data
    dp["session_factory"] = async_session_factory

    webhook_url = f"{settings.webhook_base_url}{WEBHOOK_PATH}"
    await bot.set_webhook(
        url=webhook_url,
        secret_token=settings.webhook_secret,
        allowed_updates=dp.resolve_used_update_types(),
        drop_pending_updates=True,
    )
    yield
    await bot.delete_webhook()
    await bot.session.close()

app = FastAPI(lifespan=lifespan)

@app.get("/health")
async def health_check():
    return {"status": "ok"}

@app.post(WEBHOOK_PATH)
async def telegram_webhook(request: Request):
    secret = request.headers.get("X-Telegram-Bot-Api-Secret-Token")
    if secret != settings.webhook_secret:
        return {"ok": False}
    update = Update.model_validate(await request.json(), context={"bot": bot})
    await dp.feed_update(bot, update)
    return {"ok": True}
```

```python
"""
Integration tests for the FastAPI webhook endpoint.
Covers BOT-01: /start command triggers welcome message.

These tests use httpx.AsyncClient with the FastAPI app directly.
No real Telegram API calls are made — set_webhook and feed_update are mocked.
"""
import json
from unittest.mock import AsyncMock, patch
import pytest
from httpx import AsyncClient, ASGITransport
from app.main import app, WEBHOOK_PATH
from app.config import settings


def make_start_update(user_id: int = 123456789) -> dict:
    """Build a minimal /start Update JSON matching Telegram's wire format."""
    return {
        "update_id": 1,
        "message": {
            "message_id": 1,
            "from": {
                "id": user_id,
                "is_bot": False,
                "first_name": "Test",
                "username": "testuser",
            },
            "chat": {
                "id": user_id,
                "type": "private",
                "first_name": "Test",
            },
            "date": 1700000000,
            "text": "/start",
            "entities": [{"offset": 0, "length": 6, "type": "bot_command"}],
        },
    }


@pytest.fixture
def valid_secret() -> str:
    return settings.webhook_secret


@pytest.mark.asyncio
async def test_health_check():
    """GET /health returns {"status": "ok"} — used by Railway as health probe."""
    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
        response = await client.get("/health")
    assert response.status_code == 200
    assert response.json() == {"status": "ok"}


@pytest.mark.asyncio
async def test_invalid_secret():
    """POST with wrong secret returns {"ok": false} — bot ignores the update safely."""
    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
        response = await client.post(
            WEBHOOK_PATH,
            json=make_start_update(),
            headers={"X-Telegram-Bot-Api-Secret-Token": "wrong_secret"},
        )
    assert response.status_code == 200
    assert response.json() == {"ok": False}


@pytest.mark.asyncio
async def test_start_command(valid_secret):
    """
    POST /start update with valid secret returns {"ok": true}.
    Covers BOT-01: /start is handled without error.

    Mocks:
      - bot.set_webhook / bot.delete_webhook: prevent real Telegram API calls
      - dp.feed_update: prevent actual handler execution (handler needs real DB)
      - bot.session.close: prevent cleanup error in test teardown
    """
    with (
        patch("app.main.bot.set_webhook", new_callable=AsyncMock),
        patch("app.main.bot.delete_webhook", new_callable=AsyncMock),
        patch("app.main.bot.session.close", new_callable=AsyncMock),
        patch("app.main.dp.feed_update", new_callable=AsyncMock) as mock_feed,
    ):
        async with AsyncClient(
            transport=ASGITransport(app=app), base_url="http://test"
        ) as client:
            response = await client.post(
                WEBHOOK_PATH,
                json=make_start_update(),
                headers={"X-Telegram-Bot-Api-Secret-Token": valid_secret},
            )

    assert response.status_code == 200
    assert response.json() == {"ok": True}
    mock_feed.assert_called_once()
    # Verify the update was deserialized and passed to the dispatcher
    call_args = mock_feed.call_args
    update = call_args.args[1]  # dp.feed_update(bot, update)
    assert update.update_id == 1
```
